5"""FeedParser - An email feed parser. 

6 

7The feed parser implements an interface for incrementally parsing an email 

8message, line by line. This has advantages for certain applications, such as 

9those reading email messages off a socket. 

10 

11FeedParser.feed() is the primary interface for pushing new data into the 

12parser. It returns when there's nothing more it can do with the available 

13data. When you have no more data to push into the parser, call .close(). 

14This completes the parsing and returns the root message object. 

15 

16The other advantage of this parser is that it will never raise a parsing 

17exception. Instead, when it finds something unexpected, it adds a 'defect' to 

18the current message. Defects are just instances that live on the message 

19object's .defects attribute. 

20""" 

21

31NLCRE = re.compile(r"\r\n|\r|\n") 

32NLCRE_bol = re.compile(r"(\r\n|\r|\n)") 

33NLCRE_eol = re.compile(r"(\r\n|\r|\n)\Z") 

34NLCRE_crack = re.compile(r"(\r\n|\r|\n)") 

35# RFC 2822 $3.6.8 Optional fields. ftext is %d33-57 / %d59-126, Any character 

36# except controls, SP, and ":". 

37headerRE = re.compile(r"^(From |[\041-\071\073-\176]*:|[\t ])") 

38EMPTYSTRING = "" 

39NL = "\n"

44class BufferedSubFile(object): 

45 """A file-ish object that can have new data loaded into it. 

46 

47 You can also push and pop line-matching predicates onto a stack. When the 

48 current predicate matches the current line, a false EOF response 

49 (i.e. empty string) is returned instead. This lets the parser adhere to a 

50 simple abstraction -- it parses until EOF closes the current message. 

51 """ 

52 

53 def __init__(self): 

54 # Text stream of the last partial line pushed into this object. 

55 # See issue 22233 for why this is a text stream and not a list. 

56 self._partial = StringIO(newline="") 

57 # A deque of full, pushed lines 

58 self._lines = deque() 

59 # The stack of false-EOF checking predicates. 

60 self._eofstack = [] 

61 # A flag indicating whether the file has been closed or not. 

62 self._closed = False 

63 

64 def push_eof_matcher(self, pred): 

65 self._eofstack.append(pred) 

66 

67 def pop_eof_matcher(self): 

68 return self._eofstack.pop() 

69 

70 def close(self): 

71 # Don't forget any trailing partial line. 

72 self._partial.seek(0) 

73 self.pushlines(self._partial.readlines()) 

74 self._partial.seek(0) 

75 self._partial.truncate() 

76 self._closed = True 

77 

78 def readline(self): 

79 if not self._lines: 

80 if self._closed: 

81 return "" 

82 return NeedMoreData 

83 # Pop the line off the stack and see if it matches the current 

84 # false-EOF predicate. 

85 line = self._lines.popleft() 

86 # RFC 2046, section 5.1.2 requires us to recognize outer level 

87 # boundaries at any level of inner nesting. Do this, but be sure it's 

88 # in the order of most to least nested. 

89 for ateof in reversed(self._eofstack): 

90 if ateof(line): 

91 # We're at the false EOF. But push the last line back first. 

92 self._lines.appendleft(line) 

93 return "" 

94 return line 

95 

96 def unreadline(self, line): 

97 # Let the consumer push a line back into the buffer. 

98 assert line is not NeedMoreData 

99 self._lines.appendleft(line) 

100 

101 def push(self, data): 

102 """Push some new data into this object.""" 

103 self._partial.write(data) 

104 if "\n" not in data and "\r" not in data: 

105 # No new complete lines, wait for more. 

106 return 

107 

108 # Crack into lines, preserving the linesep characters. 

109 self._partial.seek(0) 

110 parts = self._partial.readlines() 

111 self._partial.seek(0) 

112 self._partial.truncate() 

113 

114 # If the last element of the list does not end in a newline, then treat 

115 # it as a partial line. We only check for '\n' here because a line 

116 # ending with '\r' might be a line that was split in the middle of a 

117 # '\r\n' sequence (see bugs 1555570 and 1721862). 

118 if not parts[-1].endswith("\n"): 

119 self._partial.write(parts.pop()) 

120 self.pushlines(parts) 

121 

122 def pushlines(self, lines): 

123 self._lines.extend(lines) 

124 

125 def __iter__(self): 

126 return self 

127 

128 def __next__(self): 

129 line = self.readline() 

130 if line == "": 

131 raise StopIteration 

132 return line

135class FeedParser: 

136 """A feed-style parser of email.""" 

137 

138 def __init__(self, _factory=None, *, policy=compat32): 

139 """_factory is called with no arguments to create a new message obj 

140 

141 The policy keyword specifies a policy object that controls a number of 

142 aspects of the parser's operation. The default policy maintains 

143 backward compatibility. 

144 

145 """ 

146 self.policy = policy 

147 self._old_style_factory = False 

148 if _factory is None: 

149 if policy.message_factory is None: 

150 from email.message import Message 

151 

152 self._factory = Message 

153 else: 

154 self._factory = policy.message_factory 

155 else: 

156 self._factory = _factory 

157 try: 

158 _factory(policy=self.policy) 

159 except TypeError: 

160 # Assume this is an old-style factory 

161 self._old_style_factory = True 

162 self._input = BufferedSubFile() 

163 self._msgstack = [] 

164 self._parse = self._parsegen().__next__ 

165 self._cur = None 

166 self._last = None 

167 self._headersonly = False 

168 

169 # Non-public interface for supporting Parser's headersonly flag 

170 def _set_headersonly(self): 

171 self._headersonly = True 

172 

173 def feed(self, data): 

174 """Push more data into the parser.""" 

175 self._input.push(data) 

176 self._call_parse() 

177 

178 def _call_parse(self): 

179 try: 

180 self._parse() 

181 except StopIteration: 

182 pass 

183 

184 def close(self): 

185 """Parse all remaining data and return the root message object.""" 

186 self._input.close() 

187 self._call_parse() 

188 root = self._pop_message() 

189 assert not self._msgstack 

190 # Look for final set of defects 

191 if ( 

192 root.get_content_maintype() == "multipart" 

193 and not root.is_multipart() 

194 and not self._headersonly 

195 ): 

196 defect = errors.MultipartInvariantViolationDefect() 

197 self.policy.handle_defect(root, defect) 

198 return root 

199 

200 def _new_message(self): 

201 if self._old_style_factory: 

202 msg = self._factory() 

203 else: 

204 msg = self._factory(policy=self.policy) 

205 if self._cur and self._cur.get_content_type() == "multipart/digest": 

206 msg.set_default_type("message/rfc822") 

207 if self._msgstack: 

208 self._msgstack[-1].attach(msg) 

209 self._msgstack.append(msg) 

210 self._cur = msg 

211 self._last = msg 

212 

213 def _pop_message(self): 

214 retval = self._msgstack.pop() 

215 if self._msgstack: 

216 self._cur = self._msgstack[-1] 

217 else: 

218 self._cur = None 

219 return retval

221 def _parsegen(self): 

222 # Create a new message and start by parsing headers. 

223 self._new_message() 

224 headers = [] 

225 # Collect the headers, searching for a line that doesn't match the RFC 

226 # 2822 header or continuation pattern (including an empty line). 

227 for line in self._input: 

228 if line is NeedMoreData: 

229 yield NeedMoreData 

230 continue 

231 if not headerRE.match(line): 

232 # If we saw the RFC defined header/body separator 

233 # (i.e. newline), just throw it away. Otherwise the line is 

234 # part of the body so push it back. 

235 if not NLCRE.match(line): 

236 defect = errors.MissingHeaderBodySeparatorDefect() 

237 self.policy.handle_defect(self._cur, defect) 

238 self._input.unreadline(line) 

239 break 

240 headers.append(line) 

241 # Done with the headers, so parse them and figure out what we're 

242 # supposed to see in the body of the message. 

243 self._parse_headers(headers) 

244 # Headers-only parsing is a backwards compatibility hack, which was 

245 # necessary in the older parser, which could raise errors. All 

246 # remaining lines in the input are thrown into the message body. 

247 if self._headersonly: 

248 lines = [] 

249 while True: 

250 line = self._input.readline() 

251 if line is NeedMoreData: 

252 yield NeedMoreData 

253 continue 

254 if line == "": 

255 break 

256 lines.append(line) 

257 self._cur.set_payload(EMPTYSTRING.join(lines)) 

258 return 

259 if self._cur.get_content_type() == "message/delivery-status": 

260 # message/delivery-status contains blocks of headers separated by 

261 # a blank line. We'll represent each header block as a separate 

262 # nested message object, but the processing is a bit different 

263 # than standard message/* types because there is no body for the 

264 # nested messages. A blank line separates the subparts. 

265 while True: 

266 self._input.push_eof_matcher(NLCRE.match) 

267 for retval in self._parsegen(): 

268 if retval is NeedMoreData: 

269 yield NeedMoreData 

270 continue 

271 break 

272 self._pop_message() 

273 # We need to pop the EOF matcher in order to tell if we're at 

274 # the end of the current file, not the end of the last block 

275 # of message headers. 

276 self._input.pop_eof_matcher() 

277 # The input stream must be sitting at the newline or at the 

278 # EOF. We want to see if we're at the end of this subpart, so 

279 # first consume the blank line, then test the next line to see 

280 # if we're at this subpart's EOF. 

281 while True: 

282 line = self._input.readline() 

283 if line is NeedMoreData: 

284 yield NeedMoreData 

285 continue 

286 break 

287 while True: 

288 line = self._input.readline() 

289 if line is NeedMoreData: 

290 yield NeedMoreData 

291 continue 

292 break 

293 if line == "": 

294 break 

295 # Not at EOF so this is a line we're going to need. 

296 self._input.unreadline(line) 

297 return 

298 if self._cur.get_content_maintype() == "message": 

299 # The message claims to be a message/* type, then what follows is 

300 # another RFC 2822 message. 

301 for retval in self._parsegen(): 

302 if retval is NeedMoreData: 

303 yield NeedMoreData 

304 continue 

305 break 

306 self._pop_message() 

307 return 

308 if self._cur.get_content_maintype() == "multipart": 

309 boundary = self._cur.get_boundary() 

310 if boundary is None: 

311 # The message /claims/ to be a multipart but it has not 

312 # defined a boundary. That's a problem which we'll handle by 

313 # reading everything until the EOF and marking the message as 

314 # defective. 

315 defect = errors.NoBoundaryInMultipartDefect() 

316 self.policy.handle_defect(self._cur, defect) 

317 lines = [] 

318 for line in self._input: 

319 if line is NeedMoreData: 

320 yield NeedMoreData 

321 continue 

322 lines.append(line) 

323 self._cur.set_payload(EMPTYSTRING.join(lines)) 

324 return 

325 # Make sure a valid content type was specified per RFC 2045:6.4. 

326 if str(self._cur.get("content-transfer-encoding", "8bit")).lower() not in ( 

327 "7bit", 

328 "8bit", 

329 "binary", 

330 ): 

331 defect = errors.InvalidMultipartContentTransferEncodingDefect() 

332 self.policy.handle_defect(self._cur, defect) 

333 # Create a line match predicate which matches the inter-part 

334 # boundary as well as the end-of-multipart boundary. Don't push 

335 # this onto the input stream until we've scanned past the 

336 # preamble. 

337 separator = "--" + boundary 

338 boundaryre = re.compile( 

339 "(?P<sep>" 

340 + re.escape(separator) 

341 + r")(?P<end>--)?(?P<ws>[ \t]*)(?P<linesep>\r\n|\r|\n)?$" 

342 ) 

343 capturing_preamble = True 

344 preamble = [] 

345 linesep = False 

346 close_boundary_seen = False 

347 while True: 

348 line = self._input.readline() 

349 if line is NeedMoreData: 

350 yield NeedMoreData 

351 continue 

352 if line == "": 

353 break 

354 mo = boundaryre.match(line) 

355 if mo: 

356 # If we're looking at the end boundary, we're done with 

357 # this multipart. If there was a newline at the end of 

358 # the closing boundary, then we need to initialize the 

359 # epilogue with the empty string (see below). 

360 if mo.group("end"): 

361 close_boundary_seen = True 

362 linesep = mo.group("linesep") 

363 break 

364 # We saw an inter-part boundary. Were we in the preamble? 

365 if capturing_preamble: 

366 if preamble: 

367 # According to RFC 2046, the last newline belongs 

368 # to the boundary. 

369 lastline = preamble[-1] 

370 eolmo = NLCRE_eol.search(lastline) 

371 if eolmo: 

372 preamble[-1] = lastline[: -len(eolmo.group(0))] 

373 self._cur.preamble = EMPTYSTRING.join(preamble) 

374 capturing_preamble = False 

375 self._input.unreadline(line) 

376 continue 

377 # We saw a boundary separating two parts. Consume any 

378 # multiple boundary lines that may be following. Our 

379 # interpretation of RFC 2046 BNF grammar does not produce 

380 # body parts within such double boundaries. 

381 while True: 

382 line = self._input.readline() 

383 if line is NeedMoreData: 

384 yield NeedMoreData 

385 continue 

386 mo = boundaryre.match(line) 

387 if not mo: 

388 self._input.unreadline(line) 

389 break 

390 # Recurse to parse this subpart; the input stream points 

391 # at the subpart's first line. 

392 self._input.push_eof_matcher(boundaryre.match) 

393 for retval in self._parsegen(): 

394 if retval is NeedMoreData: 

395 yield NeedMoreData 

396 continue 

397 break 

398 # Because of RFC 2046, the newline preceding the boundary 

399 # separator actually belongs to the boundary, not the 

400 # previous subpart's payload (or epilogue if the previous 

401 # part is a multipart). 

402 if self._last.get_content_maintype() == "multipart": 

403 epilogue = self._last.epilogue 

404 if epilogue == "": 

405 self._last.epilogue = None 

406 elif epilogue is not None: 

407 mo = NLCRE_eol.search(epilogue) 

408 if mo: 

409 end = len(mo.group(0)) 

410 self._last.epilogue = epilogue[:-end] 

411 else: 

412 payload = self._last._payload 

413 if isinstance(payload, str): 

414 mo = NLCRE_eol.search(payload) 

415 if mo: 

416 payload = payload[: -len(mo.group(0))] 

417 self._last._payload = payload 

418 self._input.pop_eof_matcher() 

419 self._pop_message() 

420 # Set the multipart up for newline cleansing, which will 

421 # happen if we're in a nested multipart. 

422 self._last = self._cur 

423 else: 

424 # I think we must be in the preamble 

425 assert capturing_preamble 

426 preamble.append(line) 

427 # We've seen either the EOF or the end boundary. If we're still 

428 # capturing the preamble, we never saw the start boundary. Note 

429 # that as a defect and store the captured text as the payload. 

430 if capturing_preamble: 

431 defect = errors.StartBoundaryNotFoundDefect() 

432 self.policy.handle_defect(self._cur, defect) 

433 self._cur.set_payload(EMPTYSTRING.join(preamble)) 

434 epilogue = [] 

435 for line in self._input: 

436 if line is NeedMoreData: 

437 yield NeedMoreData 

438 continue 

439 self._cur.epilogue = EMPTYSTRING.join(epilogue) 

440 return 

441 # If we're not processing the preamble, then we might have seen 

442 # EOF without seeing that end boundary...that is also a defect. 

443 if not close_boundary_seen: 

444 defect = errors.CloseBoundaryNotFoundDefect() 

445 self.policy.handle_defect(self._cur, defect) 

446 return 

447 # Everything from here to the EOF is epilogue. If the end boundary 

448 # ended in a newline, we'll need to make sure the epilogue isn't 

449 # None 

450 if linesep: 

451 epilogue = [""] 

452 else: 

453 epilogue = [] 

454 for line in self._input: 

455 if line is NeedMoreData: 

456 yield NeedMoreData 

457 continue 

458 epilogue.append(line) 

459 # Any CRLF at the front of the epilogue is not technically part of 

460 # the epilogue. Also, watch out for an empty string epilogue, 

461 # which means a single newline. 

462 if epilogue: 

463 firstline = epilogue[0] 

464 bolmo = NLCRE_bol.match(firstline) 

465 if bolmo: 

466 epilogue[0] = firstline[len(bolmo.group(0)) :] 

467 self._cur.epilogue = EMPTYSTRING.join(epilogue) 

468 return 

469 # Otherwise, it's some non-multipart type, so the entire rest of the 

470 # file contents becomes the payload. 

471 lines = [] 

472 for line in self._input: 

473 if line is NeedMoreData: 

474 yield NeedMoreData 

475 continue 

476 lines.append(line) 

477 self._cur.set_payload(EMPTYSTRING.join(lines))

479 def _parse_headers(self, lines): 

480 # Passed a list of lines that make up the headers for the current msg 

481 lastheader = "" 

482 lastvalue = [] 

483 for lineno, line in enumerate(lines): 

484 # Check for continuation 

485 if line[0] in " \t": 

486 if not lastheader: 

487 # The first line of the headers was a continuation. This 

488 # is illegal, so let's note the defect, store the illegal 

489 # line, and ignore it for purposes of headers. 

490 defect = errors.FirstHeaderLineIsContinuationDefect(line) 

491 self.policy.handle_defect(self._cur, defect) 

492 continue 

493 lastvalue.append(line) 

494 continue 

495 if lastheader: 

496 self._cur.set_raw(*self.policy.header_source_parse(lastvalue)) 

497 lastheader, lastvalue = "", [] 

498 # Check for envelope header, i.e. unix-from 

499 if line.startswith("From "): 

500 if lineno == 0: 

501 # Strip off the trailing newline 

502 mo = NLCRE_eol.search(line) 

503 if mo: 

504 line = line[: -len(mo.group(0))] 

505 self._cur.set_unixfrom(line) 

506 continue 

507 elif lineno == len(lines) - 1: 

508 # Something looking like a unix-from at the end - it's 

509 # probably the first line of the body, so push back the 

510 # line and stop. 

511 self._input.unreadline(line) 

512 return 

513 else: 

514 # Weirdly placed unix-from line. Note this as a defect 

515 # and ignore it. 

516 defect = errors.MisplacedEnvelopeHeaderDefect(line) 

517 self._cur.defects.append(defect) 

518 continue 

519 # Split the line on the colon separating field name from value. 

520 # There will always be a colon, because if there wasn't the part of 

521 # the parser that calls us would have started parsing the body. 

522 i = line.find(":") 

523 

524 # If the colon is on the start of the line the header is clearly 

525 # malformed, but we might be able to salvage the rest of the 

526 # message. Track the error but keep going. 

527 if i == 0: 

528 defect = errors.InvalidHeaderDefect("Missing header name.") 

529 self._cur.defects.append(defect) 

530 continue 

531 

532 assert i > 0, "_parse_headers fed line with no : and no leading WS" 

533 lastheader = line[:i] 

534 lastvalue = [line] 

535 # Done with all the lines, so handle the last header. 

536 if lastheader: 

537 self._cur.set_raw(*self.policy.header_source_parse(lastvalue))
