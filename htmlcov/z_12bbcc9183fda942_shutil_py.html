143def _fastcopy_sendfile(fsrc, fdst): 

144 """Copy data from one regular mmap-like fd to another by using 

145 high-performance sendfile(2) syscall. 

146 This should work on Linux >= 2.6.33 only. 

147 """ 

148 # Note: copyfileobj() is left alone in order to not introduce any 

149 # unexpected breakage. Possible risks by using zero-copy calls 

150 # in copyfileobj() are: 

151 # - fdst cannot be open in "a"(ppend) mode 

152 # - fsrc and fdst may be open in "t"(ext) mode 

153 # - fsrc may be a BufferedReader (which hides unread data in a buffer), 

154 # GzipFile (which decompresses data), HTTPResponse (which decodes 

155 # chunks). 

156 # - possibly others (e.g. encrypted fs/partition?) 

157 global _USE_CP_SENDFILE 

158 try: 

159 infd = fsrc.fileno() 

160 outfd = fdst.fileno() 

161 except Exception as err: 

162 raise _GiveupOnFastCopy(err) # not a regular file 

163 

164 # Hopefully the whole file will be copied in a single call. 

165 # sendfile() is called in a loop 'till EOF is reached (0 return) 

166 # so a bufsize smaller or bigger than the actual file size 

167 # should not make any difference, also in case the file content 

168 # changes while being copied. 

169 try: 

170 blocksize = max(os.fstat(infd).st_size, 2**23) # min 8MiB 

171 except OSError: 

172 blocksize = 2**27 # 128MiB 

173 # On 32-bit architectures truncate to 1GiB to avoid OverflowError, 

174 # see bpo-38319. 

175 if sys.maxsize < 2**32: 

176 blocksize = min(blocksize, 2**30) 

177 

178 offset = 0 

179 while True: 

180 try: 

181 sent = os.sendfile(outfd, infd, offset, blocksize) 

182 except OSError as err: 

183 # ...in oder to have a more informative exception. 

184 err.filename = fsrc.name 

185 err.filename2 = fdst.name 

186 

187 if err.errno == errno.ENOTSOCK: 

188 # sendfile() on this platform (probably Linux < 2.6.33) 

189 # does not support copies between regular files (only 

190 # sockets). 

191 _USE_CP_SENDFILE = False 

192 raise _GiveupOnFastCopy(err) 

193 

194 if err.errno == errno.ENOSPC: # filesystem is full 

195 raise err from None 

196 

197 # Give up on first call and if no data was copied. 

198 if offset == 0 and os.lseek(outfd, 0, os.SEEK_CUR) == 0: 

199 raise _GiveupOnFastCopy(err) 

200 

201 raise err 

202 else: 

203 if sent == 0: 

204 break # EOF 

205 offset += sent

739# Version using fd-based APIs to protect against races 

740def _rmtree_safe_fd(stack, onexc): 

741 # Each stack item has four elements: 

742 # * func: The first operation to perform: os.lstat, os.close or os.rmdir. 

743 # Walking a directory starts with an os.lstat() to detect symlinks; in 

744 # this case, func is updated before subsequent operations and passed to 

745 # onexc() if an error occurs. 

746 # * dirfd: Open file descriptor, or None if we're processing the top-level 

747 # directory given to rmtree() and the user didn't supply dir_fd. 

748 # * path: Path of file to operate upon. This is passed to onexc() if an 

749 # error occurs. 

750 # * orig_entry: os.DirEntry, or None if we're processing the top-level 

751 # directory given to rmtree(). We used the cached stat() of the entry to 

752 # save a call to os.lstat() when walking subdirectories. 

753 func, dirfd, path, orig_entry = stack.pop() 

754 name = path if orig_entry is None else orig_entry.name 

755 try: 

756 if func is os.close: 

757 os.close(dirfd) 

758 return 

759 if func is os.rmdir: 

760 os.rmdir(name, dir_fd=dirfd) 

761 return 

762 

763 # Note: To guard against symlink races, we use the standard 

764 # lstat()/open()/fstat() trick. 

765 assert func is os.lstat 

766 if orig_entry is None: 

767 orig_st = os.lstat(name, dir_fd=dirfd) 

768 else: 

769 orig_st = orig_entry.stat(follow_symlinks=False) 

770 

771 func = os.open # For error reporting. 

772 topfd = os.open(name, os.O_RDONLY | os.O_NONBLOCK, dir_fd=dirfd) 

773 

774 func = os.path.islink # For error reporting. 

775 try: 

776 if not os.path.samestat(orig_st, os.fstat(topfd)): 

777 # Symlinks to directories are forbidden, see GH-46010. 

778 raise OSError("Cannot call rmtree on a symbolic link") 

779 stack.append((os.rmdir, dirfd, path, orig_entry)) 

780 finally: 

781 stack.append((os.close, topfd, path, orig_entry)) 

782 

783 func = os.scandir # For error reporting. 

784 with os.scandir(topfd) as scandir_it: 

785 entries = list(scandir_it) 

786 for entry in entries: 

787 fullname = os.path.join(path, entry.name) 

788 try: 

789 if entry.is_dir(follow_symlinks=False): 

790 # Traverse into sub-directory. 

791 stack.append((os.lstat, topfd, fullname, entry)) 

792 continue 

793 except OSError: 

794 pass 

795 try: 

796 os.unlink(entry.name, dir_fd=topfd) 

797 except OSError as err: 

798 onexc(os.unlink, fullname, err) 

799 except OSError as err: 

800 err.filename = path 

801 onexc(func, path, err)

811def rmtree(path, ignore_errors=False, onerror=None, *, onexc=None, dir_fd=None): 

812 """Recursively delete a directory tree. 

813 

814 If dir_fd is not None, it should be a file descriptor open to a directory; 

815 path will then be relative to that directory. 

816 dir_fd may not be implemented on your platform. 

817 If it is unavailable, using it will raise a NotImplementedError. 

818 

819 If ignore_errors is set, errors are ignored; otherwise, if onexc or 

820 onerror is set, it is called to handle the error with arguments (func, 

821 path, exc_info) where func is platform and implementation dependent; 

822 path is the argument to that function that caused it to fail; and 

823 the value of exc_info describes the exception. For onexc it is the 

824 exception instance, and for onerror it is a tuple as returned by 

825 sys.exc_info(). If ignore_errors is false and both onexc and 

826 onerror are None, the exception is reraised. 

827 

828 onerror is deprecated and only remains for backwards compatibility. 

829 If both onerror and onexc are set, onerror is ignored and onexc is used. 

830 """ 

831 

832 sys.audit("shutil.rmtree", path, dir_fd) 

833 if ignore_errors: 

834 

835 def onexc(*args): 

836 pass 

837 elif onerror is None and onexc is None: 

838 

839 def onexc(*args): 

840 raise 

841 elif onexc is None: 

842 if onerror is None: 

843 

844 def onexc(*args): 

845 raise 

846 else: 

847 # delegate to onerror 

848 def onexc(*args): 

849 func, path, exc = args 

850 if exc is None: 

851 exc_info = None, None, None 

852 else: 

853 exc_info = type(exc), exc, exc.__traceback__ 

854 return onerror(func, path, exc_info) 

855 

856 if _use_fd_functions: 

857 # While the unsafe rmtree works fine on bytes, the fd based does not. 

858 if isinstance(path, bytes): 

859 path = os.fsdecode(path) 

860 stack = [(os.lstat, dir_fd, path, None)] 

861 try: 

862 while stack: 

863 _rmtree_safe_fd(stack, onexc) 

864 finally: 

865 # Close any file descriptors still on the stack. 

866 while stack: 

867 func, fd, path, entry = stack.pop() 

868 if func is not os.close: 

869 continue 

870 try: 

871 os.close(fd) 

872 except OSError as err: 

873 onexc(os.close, path, err) 

874 else: 

875 if dir_fd is not None: 

876 raise NotImplementedError("dir_fd unavailable on this platform") 

877 try: 

878 if _rmtree_islink(path): 

879 # symlinks to directories are forbidden, see bug #1669 

880 raise OSError("Cannot call rmtree on a symbolic link") 

881 except OSError as err: 

882 onexc(os.path.islink, path, err) 

883 # can't continue even if onexc hook returns 

884 return 

885 return _rmtree_unsafe(path, onexc)

890rmtree.avoids_symlink_attacks = _use_fd_functions

912def move(src, dst, copy_function=copy2): 

913 """Recursively move a file or directory to another location. This is 

914 similar to the Unix "mv" command. Return the file or directory's 

915 destination. 

916 

917 If dst is an existing directory or a symlink to a directory, then src is 

918 moved inside that directory. The destination path in that directory must 

919 not already exist. 

920 

921 If dst already exists but is not a directory, it may be overwritten 

922 depending on os.rename() semantics. 

923 

924 If the destination is on our current filesystem, then rename() is used. 

925 Otherwise, src is copied to the destination and then removed. Symlinks are 

926 recreated under the new name if os.rename() fails because of cross 

927 filesystem renames. 

928 

929 The optional `copy_function` argument is a callable that will be used 

930 to copy the source or it will be delegated to `copytree`. 

931 By default, copy2() is used, but any function that supports the same 

932 signature (like copy()) can be used. 

933 

934 A lot more could be done here... A look at a mv.c shows a lot of 

935 the issues this implementation glosses over. 

936 

937 """ 

938 sys.audit("shutil.move", src, dst) 

939 real_dst = dst 

940 if os.path.isdir(dst): 

941 if _samefile(src, dst) and not os.path.islink(src): 

942 # We might be on a case insensitive filesystem, 

943 # perform the rename anyway. 

944 os.rename(src, dst) 

945 return 

946 

947 # Using _basename instead of os.path.basename is important, as we must 

948 # ignore any trailing slash to avoid the basename returning '' 

949 real_dst = os.path.join(dst, _basename(src)) 

950 

951 if os.path.exists(real_dst): 

952 raise Error("Destination path '%s' already exists" % real_dst) 

953 try: 

954 os.rename(src, real_dst) 

955 except OSError: 

956 if os.path.islink(src): 

957 linkto = os.readlink(src) 

958 os.symlink(linkto, real_dst) 

959 os.unlink(src) 

960 elif os.path.isdir(src): 

961 if _destinsrc(src, dst): 

962 raise Error( 

963 "Cannot move a directory '%s' into itself '%s'." % (src, dst) 

964 ) 

965 if _is_immutable(src) or ( 

966 not os.access(src, os.W_OK) 

967 and os.listdir(src) 

968 and sys.platform == "darwin" 

969 ): 

970 raise PermissionError( 

971 "Cannot move the non-empty directory " 

972 "'%s': Lacking write permission to '%s'." % (src, src) 

973 ) 

974 copytree(src, real_dst, copy_function=copy_function, symlinks=True) 

975 rmtree(src) 

976 else: 

977 copy_function(src, real_dst) 

978 os.unlink(src) 

979 return real_dst

1584def get_terminal_size(fallback=(80, 24)): 

1585 """Get the size of the terminal window. 

1586

1588 and LINES respectively, is checked. If the variable is defined and

1596 the system doesn't support querying, or because we are not 

1597 connected to a terminal, the value given in fallback parameter 

1598 is used. Fallback defaults to (80, 24) which is the default 

1599 size used by many terminal emulators. 

1600 

1601 The value returned is a named tuple of type os.terminal_size. 

1602 """

1604 try: 

1605 columns = int(os.environ["COLUMNS"]) 

1606 except (KeyError, ValueError): 

1607 columns = 0 

1608 

1609 try: 

1610 lines = int(os.environ["LINES"]) 

1611 except (KeyError, ValueError): 

1612 lines = 0 

1613 

1614 # only query if necessary 

1615 if columns <= 0 or lines <= 0: 

1616 try: 

1617 size = os.get_terminal_size(sys.__stdout__.fileno()) 

1618 except (AttributeError, ValueError, OSError): 

1619 # stdout is None, closed, detached, or not a terminal, or 

1620 # os.get_terminal_size() is unsupported 

1621 size = os.terminal_size(fallback)

1623 columns = size.columns or fallback[0] 

1624 if lines <= 0:

1648def which(cmd, mode=os.F_OK | os.X_OK, path=None): 

1649 """Given a command, mode, and a PATH string, return the path which 

1650 conforms to the given mode on the PATH, or None if there is no such 

1651 file. 

1652 

1653 `mode` defaults to os.F_OK | os.X_OK. `path` defaults to the result 

1654 of os.environ.get("PATH"), or can be overridden with a custom search 

1655 path. 

1656 

1657 """ 

1658 use_bytes = isinstance(cmd, bytes) 

1659 

1660 # If we're given a path with a directory part, look it up directly rather 

1661 # than referring to PATH directories. This includes checking relative to 

1662 # the current directory, e.g. ./script 

1663 dirname, cmd = os.path.split(cmd) 

1664 if dirname: 

1665 path = [dirname] 

1666 else: 

1667 if path is None: 

1668 path = os.environ.get("PATH", None) 

1669 if path is None: 

1670 try: 

1671 path = os.confstr("CS_PATH") 

1672 except (AttributeError, ValueError): 

1673 # os.confstr() or CS_PATH is not available 

1674 path = os.defpath 

1675 # bpo-35755: Don't use os.defpath if the PATH environment variable 

1676 # is set to an empty string 

1677 

1678 # PATH='' doesn't match, whereas PATH=':' looks in the current 

1679 # directory 

1680 if not path: 

1681 return None 

1682 

1683 if use_bytes: 

1684 path = os.fsencode(path) 

1685 path = path.split(os.fsencode(os.pathsep)) 

1686 else: 

1687 path = os.fsdecode(path) 

1688 path = path.split(os.pathsep) 

1689 

1690 if sys.platform == "win32" and _win_path_needs_curdir(cmd, mode): 

1691 curdir = os.curdir 

1692 if use_bytes: 

1693 curdir = os.fsencode(curdir) 

1694 path.insert(0, curdir) 

1695 

1696 if sys.platform == "win32": 

1697 # PATHEXT is necessary to check on Windows. 

1698 pathext_source = os.getenv("PATHEXT") or _WIN_DEFAULT_PATHEXT 

1699 pathext = pathext_source.split(os.pathsep) 

1700 pathext = [ext.rstrip(".") for ext in pathext if ext] 

1701 

1702 if use_bytes: 

1703 pathext = [os.fsencode(ext) for ext in pathext] 

1704 

1705 files = [cmd + ext for ext in pathext] 

1706 

1707 # If X_OK in mode, simulate the cmd.exe behavior: look at direct 

1708 # match if and only if the extension is in PATHEXT. 

1709 # If X_OK not in mode, simulate the first result of where.exe: 

1710 # always look at direct match before a PATHEXT match. 

1711 normcmd = cmd.upper() 

1712 if not (mode & os.X_OK) or any( 

1713 normcmd.endswith(ext.upper()) for ext in pathext 

1714 ): 

1715 files.insert(0, cmd) 

1716 else: 

1717 # On other platforms you don't have things like PATHEXT to tell you 

1718 # what file suffixes are executable, so just pass on cmd as-is. 

1719 files = [cmd] 

1720 

1721 seen = set() 

1722 for dir in path: 

1723 normdir = os.path.normcase(dir) 

1724 if normdir not in seen: 

1725 seen.add(normdir) 

1726 for thefile in files: 

1727 name = os.path.join(dir, thefile) 

1728 if _access_check(name, mode): 

1729 return name 

1730 return None
