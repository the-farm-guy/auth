18class _PolicyBase: 

19 """Policy Object basic framework. 

20 

21 This class is useless unless subclassed. A subclass should define 

22 class attributes with defaults for any values that are to be 

23 managed by the Policy object. The constructor will then allow 

24 non-default values to be set for these attributes at instance 

25 creation time. The instance will be callable, taking these same 

26 attributes keyword arguments, and returning a new instance 

27 identical to the called instance except for those values changed 

28 by the keyword arguments. Instances may be added, yielding new 

29 instances with any non-default values from the right hand 

30 operand overriding those in the left hand operand. That is, 

31 

32 A + B == A(<non-default values of B>) 

33 

34 The repr of an instance can be used to reconstruct the object 

35 if and only if the repr of the values can be used to reconstruct 

36 those values. 

37 

38 """

115class Policy(_PolicyBase, metaclass=abc.ABCMeta): 

116 r"""Controls for how messages are interpreted and formatted. 

117 

118 Most of the classes and many of the methods in the email package accept 

119 Policy objects as parameters. A Policy object contains a set of values and 

120 functions that control how input is interpreted and how output is rendered. 

121 For example, the parameter 'raise_on_defect' controls whether or not an RFC 

122 violation results in an error being raised or not, while 'max_line_length' 

123 controls the maximum length of output lines when a Message is serialized. 

124 

125 Any valid attribute may be overridden when a Policy is created by passing 

126 it as a keyword argument to the constructor. Policy objects are immutable, 

127 but a new Policy object can be created with only certain values changed by 

128 calling the Policy instance with keyword arguments. Policy objects can 

129 also be added, producing a new Policy object in which the non-default 

130 attributes set in the right hand operand overwrite those specified in the 

131 left operand. 

132 

133 Settable attributes: 

134 

135 raise_on_defect -- If true, then defects should be raised as errors. 

136 Default: False. 

137 

138 linesep -- string containing the value to use as separation 

139 between output lines. Default '\n'. 

140 

141 cte_type -- Type of allowed content transfer encodings 

142 

143 7bit -- ASCII only 

144 8bit -- Content-Transfer-Encoding: 8bit is allowed 

145 

146 Default: 8bit. Also controls the disposition of 

147 (RFC invalid) binary data in headers; see the 

148 documentation of the binary_fold method. 

149 

150 max_line_length -- maximum length of lines, excluding 'linesep', 

151 during serialization. None or 0 means no line 

152 wrapping is done. Default is 78. 

153 

154 mangle_from_ -- a flag that, when True escapes From_ lines in the 

155 body of the message by putting a `>' in front of 

156 them. This is used when the message is being 

157 serialized by a generator. Default: False. 

158 

159 message_factory -- the class to use to create new message objects. 

160 If the value is None, the default is Message. 

161 

162 verify_generated_headers 

163 -- if true, the generator verifies that each header 

164 they are properly folded, so that a parser won't 

165 treat it as multiple headers, start-of-body, or 

166 part of another header. 

167 This is a check against custom Header & fold() 

168 implementations. 

169 """ 

170 

171 raise_on_defect = False 

172 linesep = "\n" 

173 cte_type = "8bit" 

174 max_line_length = 78 

175 mangle_from_ = False 

176 message_factory = None 

177 verify_generated_headers = True

179 def handle_defect(self, obj, defect): 

180 """Based on policy, either raise defect or call register_defect. 

181 

182 handle_defect(obj, defect) 

183 

184 defect should be a Defect subclass, but in any case must be an 

185 Exception subclass. obj is the object on which the defect should be 

186 registered if it is not raised. If the raise_on_defect is True, the 

187 defect is raised as an error, otherwise the object and the defect are 

188 passed to register_defect. 

189 

190 This method is intended to be called by parsers that discover defects. 

191 The email package parsers always call it with Defect instances. 

192 

193 """ 

194 if self.raise_on_defect: 

195 raise defect 

196 self.register_defect(obj, defect) 

197 

198 def register_defect(self, obj, defect): 

199 """Record 'defect' on 'obj'. 

200 

201 Called by handle_defect if raise_on_defect is False. This method is 

202 part of the Policy API so that Policy subclasses can implement custom 

203 defect handling. The default implementation calls the append method of 

204 the defects attribute of obj. The objects used by the email package by 

205 default that get passed to this method will always have a defects 

206 attribute with an append method. 

207 

208 """ 

209 obj.defects.append(defect)

211 def header_max_count(self, name): 

212 """Return the maximum allowed number of headers named 'name'. 

213 

214 Called when a header is added to a Message object. If the returned 

215 value is not 0 or None, and there are already a number of headers with 

216 the name 'name' equal to the value returned, a ValueError is raised. 

217 

218 Because the default behavior of Message's __setitem__ is to append the 

219 value to the list of headers, it is easy to create duplicate headers 

220 without realizing it. This method allows certain headers to be limited 

221 in the number of instances of that header that may be added to a 

222 Message programmatically. (The limit is not observed by the parser, 

223 which will faithfully produce as many headers as exist in the message 

224 being parsed.) 

225 

226 The default implementation returns None for all header names. 

227 """ 

228 return None

282class Compat32(Policy): 

283 """+ 

284 This particular policy is the backward compatibility Policy. It 

285 replicates the behavior of the email package version 5.1.

290 def _sanitize_header(self, name, value): 

291 # If the header value contains surrogates, return a Header using 

292 # the unknown-8bit charset to encode the bytes as encoded words. 

293 if not isinstance(value, str): 

294 # Assume it is already a header object 

295 return value 

296 if _has_surrogates(value): 

297 return header.Header(value, charset=_charset.UNKNOWN8BIT, header_name=name) 

298 else: 

299 return value

301 def header_source_parse(self, sourcelines): 

302 """+ 

303 The name is parsed as everything up to the ':' and returned unmodified. 

304 The value is determined by stripping leading whitespace off the 

305 remainder of the first line joined with all subsequent lines, and 

306 stripping any trailing carriage return or linefeed characters. 

307 

308 """ 

309 name, value = sourcelines[0].split(":", 1) 

310 value = "".join((value, *sourcelines[1:])).lstrip(" \t\r\n") 

311 return (name, value.rstrip("\r\n")) 

312 

313 def header_store_parse(self, name, value): 

314 """+ 

315 The name and value are returned unmodified. 

316 """ 

317 return (name, value) 

318 

319 def header_fetch_parse(self, name, value): 

320 """+ 

321 If the value contains binary data, it is converted into a Header object 

322 using the unknown-8bit charset. Otherwise it is returned unmodified. 

323 """ 

324 return self._sanitize_header(name, value)

326 def fold(self, name, value): 

327 """+ 

328 Headers are folded using the Header folding algorithm, which preserves 

329 existing line breaks in the value, and wraps each resulting line to the 

330 max_line_length. Non-ASCII binary data are CTE encoded using the 

331 unknown-8bit charset. 

332 

333 """ 

334 return self._fold(name, value, sanitize=True) 

335 

336 def fold_binary(self, name, value): 

337 """+ 

338 Headers are folded using the Header folding algorithm, which preserves 

339 existing line breaks in the value, and wraps each resulting line to the 

340 max_line_length. If cte_type is 7bit, non-ascii binary data is CTE 

341 encoded using the unknown-8bit charset. Otherwise the original source 

342 header is used, with its existing line breaks and/or binary data. 

343 

344 """ 

345 folded = self._fold(name, value, sanitize=self.cte_type == "7bit") 

346 return folded.encode("ascii", "surrogateescape")

348 def _fold(self, name, value, sanitize): 

349 parts = [] 

350 parts.append("%s: " % name) 

351 if isinstance(value, str): 

352 if _has_surrogates(value): 

353 if sanitize: 

354 h = header.Header( 

355 value, charset=_charset.UNKNOWN8BIT, header_name=name 

356 ) 

357 else: 

358 # If we have raw 8bit data in a byte string, we have no idea 

359 # what the encoding is. There is no safe way to split this 

360 # string. If it's ascii-subset, then we could do a normal 

361 # ascii split, but if it's multibyte then we could break the 

362 # string. There's no way to know so the least harm seems to 

363 # be to not split the string and risk it being too long. 

364 parts.append(value) 

365 h = None 

366 else: 

367 h = header.Header(value, header_name=name) 

368 else: 

369 # Assume it is a Header-like object. 

370 h = value 

371 if h is not None: 

372 # The Header class interprets a value of None for maxlinelen as the 

373 # default value of 78, as recommended by RFC 2822. 

374 maxlinelen = 0 

375 if self.max_line_length is not None: 

376 maxlinelen = self.max_line_length 

377 parts.append(h.encode(linesep=self.linesep, maxlinelen=maxlinelen)) 

378 parts.append(self.linesep) 

379 return "".join(parts)
