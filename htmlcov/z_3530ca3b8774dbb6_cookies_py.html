39r""" 

40Here's a sample session to show how to use this module. 

41At the moment, this is the only documentation. 

42 

43The Basics 

44---------- 

45 

46Importing is easy... 

47 

48 >>> from http import cookies 

49 

50Most of the time you start by creating a cookie. 

51 

52 >>> C = cookies.SimpleCookie() 

53 

54Once you've created your Cookie, you can add values just as if it were 

55a dictionary. 

56 

57 >>> C = cookies.SimpleCookie() 

58 >>> C["fig"] = "newton" 

59 >>> C["sugar"] = "wafer" 

60 >>> C.output() 

61 'Set-Cookie: fig=newton\r\nSet-Cookie: sugar=wafer' 

62 

63Notice that the printable representation of a Cookie is the 

64appropriate format for a Set-Cookie: header. This is the 

65default behavior. You can change the header and printed 

66attributes by using the .output() function 

67 

68 >>> C = cookies.SimpleCookie() 

69 >>> C["rocky"] = "road" 

70 >>> C["rocky"]["path"] = "/cookie" 

71 >>> print(C.output(header="Cookie:")) 

72 Cookie: rocky=road; Path=/cookie 

73 >>> print(C.output(attrs=[], header="Cookie:")) 

74 Cookie: rocky=road 

75 

76The load() method of a Cookie extracts cookies from a string. In a 

77CGI script, you would use this method to extract the cookies from the 

78HTTP_COOKIE environment variable. 

79 

80 >>> C = cookies.SimpleCookie() 

81 >>> C.load("chips=ahoy; vienna=finger") 

82 >>> C.output() 

83 'Set-Cookie: chips=ahoy\r\nSet-Cookie: vienna=finger' 

84 

85The load() method is darn-tootin smart about identifying cookies 

86within a string. Escaped quotation marks, nested semicolons, and other 

87such trickeries do not confuse it. 

88 

89 >>> C = cookies.SimpleCookie() 

90 >>> C.load('keebler="E=everybody; L=\\"Loves\\"; fudge=\\012;";') 

91 >>> print(C) 

92 Set-Cookie: keebler="E=everybody; L=\"Loves\"; fudge=\012;" 

93 

94Each element of the Cookie also supports all of the RFC 2109 

95Cookie attributes. Here's an example which sets the Path 

96attribute. 

97 

98 >>> C = cookies.SimpleCookie() 

99 >>> C["oreo"] = "doublestuff" 

100 >>> C["oreo"]["path"] = "/" 

101 >>> print(C) 

102 Set-Cookie: oreo=doublestuff; Path=/ 

103 

104Each dictionary element has a 'value' attribute, which gives you 

105back the value associated with the key. 

106 

107 >>> C = cookies.SimpleCookie() 

108 >>> C["twix"] = "none for you" 

109 >>> C["twix"].value 

110 'none for you' 

111 

112The SimpleCookie expects that all values should be standard strings. 

113Just to be sure, SimpleCookie invokes the str() builtin to convert 

114the value to a string, when the values are set dictionary-style. 

115 

116 >>> C = cookies.SimpleCookie() 

117 >>> C["number"] = 7 

118 >>> C["string"] = "seven" 

119 >>> C["number"].value 

120 '7' 

121 >>> C["string"].value 

122 'seven' 

123 >>> C.output() 

124 'Set-Cookie: number=7\r\nSet-Cookie: string=seven' 

125 

126Finis. 

127"""

150# These quoting routines conform to the RFC2109 specification, which in 

151# turn references the character definitions from RFC2068. They provide 

152# a two-way quoting algorithm. Any non-text character is translated 

153# into a 4 character sequence: a forward-slash followed by the 

154# three-digit octal equivalent of the character. Any '\' or '"' is 

155# quoted with a preceding '\' slash. 

156# Because of the way browsers really handle cookies (as opposed to what 

157# the RFC says) we also encode "," and ";". 

158# 

159# These are taken from RFC2068 and RFC2109. 

160# _LegalChars is the list of chars which don't require "'s 

161# _Translator hash-table for fast quoting 

162# 

163_LegalChars = string.ascii_letters + string.digits + "!#$%&'*+-.^_`|~:" 

164_UnescapedChars = _LegalChars + " ()/<=>?@[]{}" 

165 

166_Translator = { 

167 n: "\\%03o" % n for n in set(range(256)) - set(map(ord, _UnescapedChars)) 

168} 

169_Translator.update( 

170 { 

171 ord('"'): '\\"', 

172 ord("\\"): "\\\\", 

173 } 

174) 

175 

176_is_legal_key = re.compile("[%s]+" % re.escape(_LegalChars)).fullmatch

179def _quote(str): 

180 r"""Quote a string for use in a cookie header. 

181 

182 If the string does not need to be double-quoted, then just return the 

183 string. Otherwise, surround the string in doublequotes and quote 

184 (with a \) special characters. 

185 """ 

186 if str is None or _is_legal_key(str): 

187 return str 

188 else: 

189 return '"' + str.translate(_Translator) + '"' 

190 

191 

192_unquote_sub = re.compile(r"\\(?:([0-3][0-7][0-7])|(.))").sub 

193

195def _unquote_replace(m): 

196 if m[1]: 

197 return chr(int(m[1], 8)) 

198 else: 

199 return m[2]

202def _unquote(str): 

203 # If there aren't any doublequotes, 

204 # then there can't be any special characters. See RFC 2109. 

205 if str is None or len(str) < 2: 

206 return str 

207 if str[0] != '"' or str[-1] != '"': 

208 return str 

209 

210 # We have to assume that we must decode this string. 

211 # Down to work. 

212 

213 # Remove the "s 

214 str = str[1:-1] 

215 

216 # Check for special sequences. Examples: 

217 # \012 --> \n 

218 # \" --> " 

219 # 

220 return _unquote_sub(_unquote_replace, str)

249def _getdate(future=0, weekdayname=_weekdayname, monthname=_monthname): 

250 from time import gmtime, time 

251 

252 now = time() 

253 year, month, day, hh, mm, ss, wd, y, z = gmtime(now + future) 

254 return "%s, %02d %3s %4d %02d:%02d:%02d GMT" % ( 

255 weekdayname[wd], 

256 day, 

257 monthname[month], 

258 year, 

259 hh,

265class Morsel(dict): 

266 """A class to hold ONE (key, value) pair. 

267 

268 In a cookie, each such pair may have several attributes, so this class is 

269 used to keep the attributes associated with the appropriate key,value pair. 

270 This class also includes a coded_value attribute, which is used to hold 

271 the network representation of the value. 

272 """ 

273 

274 # RFC 2109 lists these attributes as reserved: 

275 # path comment domain 

276 # max-age secure version 

277 # 

278 # For historical reasons, these attributes are also reserved: 

279 # expires 

280 # 

281 # This is an extension from Microsoft: 

282 # httponly 

283 # 

284 # This dictionary provides a mapping from the lowercase 

285 # variant on the left to the appropriate traditional 

286 # formatting on the right. 

287 _reserved = { 

288 "expires": "expires", 

289 "path": "Path", 

290 "comment": "Comment", 

291 "domain": "Domain", 

292 "max-age": "Max-Age", 

293 "secure": "Secure", 

294 "httponly": "HttpOnly", 

295 "version": "Version", 

296 "samesite": "SameSite", 

297 } 

298 

299 _flags = {"secure", "httponly"} 

300 

301 def __init__(self): 

302 # Set defaults 

303 self._key = self._value = self._coded_value = None 

304 

305 # Set default attributes 

306 for key in self._reserved: 

307 dict.__setitem__(self, key, "") 

308 

309 @property 

310 def key(self): 

311 return self._key 

312 

313 @property 

314 def value(self): 

315 return self._value 

316 

317 @property 

318 def coded_value(self): 

319 return self._coded_value 

320 

321 def __setitem__(self, K, V): 

322 K = K.lower() 

323 if not K in self._reserved: 

324 raise CookieError("Invalid attribute %r" % (K,)) 

325 dict.__setitem__(self, K, V) 

326 

327 def setdefault(self, key, val=None): 

328 key = key.lower() 

329 if key not in self._reserved: 

330 raise CookieError("Invalid attribute %r" % (key,)) 

331 return dict.setdefault(self, key, val) 

332 

333 def __eq__(self, morsel): 

334 if not isinstance(morsel, Morsel): 

335 return NotImplemented 

336 return ( 

337 dict.__eq__(self, morsel) 

338 and self._value == morsel._value 

339 and self._key == morsel._key 

340 and self._coded_value == morsel._coded_value 

341 ) 

342 

343 __ne__ = object.__ne__ 

344 

345 def copy(self): 

346 morsel = Morsel() 

347 dict.update(morsel, self) 

348 morsel.__dict__.update(self.__dict__) 

349 return morsel 

350 

351 def update(self, values): 

352 data = {} 

353 for key, val in dict(values).items(): 

354 key = key.lower() 

355 if key not in self._reserved: 

356 raise CookieError("Invalid attribute %r" % (key,)) 

357 data[key] = val 

358 dict.update(self, data) 

359 

360 def isReservedKey(self, K): 

361 return K.lower() in self._reserved 

362 

363 def set(self, key, val, coded_val): 

364 if key.lower() in self._reserved: 

365 raise CookieError("Attempt to set a reserved key %r" % (key,)) 

366 if not _is_legal_key(key): 

367 raise CookieError("Illegal key %r" % (key,)) 

368 

369 # It's a good key, so save it. 

370 self._key = key 

371 self._value = val 

372 self._coded_value = coded_val 

373 

374 def __getstate__(self): 

375 return { 

376 "key": self._key, 

377 "value": self._value, 

378 "coded_value": self._coded_value, 

379 } 

380 

381 def __setstate__(self, state): 

382 self._key = state["key"] 

383 self._value = state["value"] 

384 self._coded_value = state["coded_value"]

396 return """ 

397 <script type="text/javascript"> 

398 <!-- begin hiding 

399 document.cookie = \"%s\"; 

400 // end hiding --> 

401 </script> 

402 """ % (self.OutputString(attrs).replace('"', r"\""))

404 def OutputString(self, attrs=None): 

405 # Build up our result 

406 # 

407 result = [] 

408 append = result.append 

409 

410 # First, the key=value pair 

411 append("%s=%s" % (self.key, self.coded_value)) 

412 

413 # Now add any defined attributes 

414 if attrs is None: 

415 attrs = self._reserved 

416 items = sorted(self.items()) 

417 for key, value in items: 

418 if value == "": 

419 continue 

420 if key not in attrs: 

421 continue 

422 if key == "expires" and isinstance(value, int): 

423 append("%s=%s" % (self._reserved[key], _getdate(value))) 

424 elif key == "max-age" and isinstance(value, int): 

425 append("%s=%d" % (self._reserved[key], value)) 

426 elif key == "comment" and isinstance(value, str): 

427 append("%s=%s" % (self._reserved[key], _quote(value))) 

428 elif key in self._flags: 

429 if value: 

430 append(str(self._reserved[key])) 

431 else: 

432 append("%s=%s" % (self._reserved[key], value)) 

433 

434 # Return the result 

435 return _semispacejoin(result)

440# 

441# Pattern for finding cookie 

442# 

443# This used to be strict parsing based on the RFC2109 and RFC2068 

444# specifications. I have since discovered that MSIE 3.0x doesn't 

445# follow the character rules outlined in those specs. As a 

446# result, the parsing rules here are less strict. 

447# 

448 

449_LegalKeyChars = r"\w\d!#%&'~_`><@,:/\$\*\+\-\.\^\|\)\(\?\}\{\=" 

450_LegalValueChars = _LegalKeyChars + r"\[\]" 

451_CookiePattern = re.compile( 

452 r""" 

453 \s* # Optional whitespace at start of cookie 

454 (?P<key> # Start of group 'key' 

455 [""" 

456 + _LegalKeyChars 

457 + r"""]+? # Any word of at least one letter 

458 ) # End of group 'key' 

459 ( # Optional group: there may not be a value. 

460 \s*=\s* # Equal Sign 

461 (?P<val> # Start of group 'val' 

462 "(?:[^\\"]|\\.)*" # Any double-quoted string 

463 | # or 

464 # Special case for "expires" attr 

465 (\w{3,6}day|\w{3}),\s # Day of the week or abbreviated day 

466 [\w\d\s-]{9,11}\s[\d:]{8}\sGMT # Date and time in specific format 

467 | # or 

468 [""" 

469 + _LegalValueChars 

470 + r"""]* # Any word or empty string 

471 ) # End of group 'val' 

472 )? # End of optional value group 

473 \s* # Any number of spaces. 

474 (\s+|;|$) # Ending either at space, semicolon, or EOS. 

475 """, 

476 re.ASCII | re.VERBOSE, 

477) # re.ASCII may be removed if safe.

548 def load(self, rawdata): 

549 """Load cookies from a string (presumably HTTP_COOKIE) or 

550 from a dictionary. Loading cookies from a dictionary 'd' 

551 is equivalent to calling: 

552 map(Cookie.__setitem__, d.keys(), d.values()) 

553 """ 

554 if isinstance(rawdata, str): 

555 self.__parse_string(rawdata) 

556 else: 

557 # self.update() wouldn't call our custom __setitem__ 

558 for key, value in rawdata.items(): 

559 self[key] = value 

560 return 

561 

562 def __parse_string(self, str, patt=_CookiePattern): 

563 i = 0 # Our starting point 

564 n = len(str) # Length of string 

565 parsed_items = [] # Parsed (type, key, value) triples 

566 morsel_seen = False # A key=value pair was previously encountered 

567 

568 TYPE_ATTRIBUTE = 1 

569 TYPE_KEYVALUE = 2 

570 

571 # We first parse the whole cookie string and reject it if it's 

572 # syntactically invalid (this helps avoid some classes of injection 

573 # attacks). 

574 while 0 <= i < n: 

575 # Start looking for a cookie 

576 match = patt.match(str, i) 

577 if not match: 

578 # No more cookies 

579 break 

580 

581 key, value = match.group("key"), match.group("val") 

582 i = match.end(0) 

583 

584 if key[0] == "$": 

585 if not morsel_seen: 

586 # We ignore attributes which pertain to the cookie 

587 # mechanism as a whole, such as "$Version". 

588 # See RFC 2965. (Does anyone care?) 

589 continue 

590 parsed_items.append((TYPE_ATTRIBUTE, key[1:], value)) 

591 elif key.lower() in Morsel._reserved: 

592 if not morsel_seen: 

593 # Invalid cookie string 

594 return 

595 if value is None: 

596 if key.lower() in Morsel._flags: 

597 parsed_items.append((TYPE_ATTRIBUTE, key, True)) 

598 else: 

599 # Invalid cookie string 

600 return 

601 else: 

602 parsed_items.append((TYPE_ATTRIBUTE, key, _unquote(value))) 

603 elif value is not None: 

604 parsed_items.append((TYPE_KEYVALUE, key, self.value_decode(value))) 

605 morsel_seen = True 

606 else: 

607 # Invalid cookie string 

608 return 

609 

610 # The cookie string is valid, apply it. 

611 M = None # current morsel 

612 for tp, key, value in parsed_items: 

613 if tp == TYPE_ATTRIBUTE: 

614 assert M is not None 

615 M[key] = value 

616 else: 

617 assert tp == TYPE_KEYVALUE 

618 rval, cval = value 

619 self.__set(key, rval, cval) 

620 M = self[key]
